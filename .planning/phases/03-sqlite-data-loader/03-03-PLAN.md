    2. Update handleLoadFile() to detect file type and call appropriate loader:
       ```typescript
       const handleLoadFile = async (file: File, tableName?: string) => {
         isLoading.value = true
         try {
           if (file.name.endsWith('.json')) {
             const pointData = await DataProvider.loadFromFile(file)
             pointData.value = pointData
           } else if (file.name.endsWith('.db') || file.name.endsWith('.sqlite')) {
             const result = await DataProvider.loadSqliteFile(file, tableName)
             pointData.value = result.pointData
           }
           loadError.value = null
         } catch (error) {
           console.error('Error loading file:', error)
           loadError.value = error instanceof Error ? error.message : 'Error loading file'
           // Preserve existing pointData on load failure (CONTEXT.md)
         } finally {
           isLoading.value = false
         }
       }
       ```

    3. Update DataLoadControl template binding:
       - Add @table-selected event handler: `@table-selected="handleTableSelected"`
       - Pass isLoading prop to DataLoadControl: `:is-loading="isLoading"`
       - Pass file to DataLoadControl: :file="currentFile" (store current file for table selection)

    4. Add handleTableSelected() method:
       ```typescript
       const handleTableSelected = (tableName: string) => {
         if (currentFile.value) {
           handleLoadFile(currentFile.value, tableName)
         }
       }
       ```

    5. Add currentFile ref to track loaded file:
       ```typescript
       const currentFile = ref<File | null>(null)
       ```

    6. Update DataLoadControl's file select handler to set currentFile:
       - Modify DataLoadControl to emit 'table-selected' event with tableName when user selects table
       - Store currentFile for table selection
